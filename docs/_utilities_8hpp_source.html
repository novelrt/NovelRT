97 [[nodiscard]] inline constexpr int32_t PopCount(uint32_t value) noexcept
98 {
99#if __cpp_lib_bitops
100 return std::popcount(value);
101#else
102 const uint32_t c1 = 0x55555555u;
103 const uint32_t c2 = 0x33333333u;
104 const uint32_t c3 = 0x0F0F0F0Fu;
105 const uint32_t c4 = 0x01010101u;
106
107 value -= (value >> 1) & c1;
108 value = (value & c2) + ((value >> 2) & c2);
109 value = (((value + (value >> 4)) & c3) * c4) >> 24;
110
111 return static_cast<int32_t>(value);
112#endif
113 }

127 [[nodiscard]] inline constexpr uint32_t Log2(uint32_t value) noexcept
128 {
129 value |= value >> 1;

131 value |= value >> 4;

148 [[nodiscard]] inline constexpr uint32_t LeadingZeroCount32(uint32_t value) noexcept
149 {
150#if __cpp_lib_bitops
151 return static_cast<uint32_t>(std::countl_zero(value));
152#else
153 if (value == 0)

159 return 31 ^ Log2(value);

174 [[nodiscard]] inline constexpr uint64_t LeadingZeroCount64(uint64_t value) noexcept
175 {
176#if __cpp_lib_bitops
177 return static_cast<uint64_t>(std::countl_zero(value));
178#else
179 uint32_t hi = static_cast<uint32_t>(value >> 32);
180
181 if (hi == 0)
182 {
183 return 32 + LeadingZeroCount32(static_cast<uint32_t>(value));
184 }
185 else
186 {
187 return LeadingZeroCount32(hi);
188 }
189#endif
190 }
