143 void operator()(SystemScheduler& scheduler)

161 EcsGraphicsBuilder& WithDefaultSpriteRendering()
162 {
163 typename SpriteRendererSystem<TGraphicsBackend>::SpritePass pass{};
164
165 ConfigureRenderPasses(
166 [&](RenderPassManager<TGraphicsBackend>& renderPassManager)
167 {
168 NovelRT::Graphics::GraphicsRenderPassDescription passDesc{};
169 NovelRT::Graphics::GraphicsAttachmentDescription attachmentDesc{};
170
171 attachmentDesc.texelFormat = _graphicsDevice->GetSwapchain()->GetFormat();
172 attachmentDesc.loadOp = NovelRT::Graphics::LoadOp::Clear;
173 attachmentDesc.storeOp = NovelRT::Graphics::StoreOp::Store;
174 attachmentDesc.initialLayout = NovelRT::Graphics::ImageLayout::Undefined;
175 attachmentDesc.finalLayout = NovelRT::Graphics::ImageLayout::Present;
176
177 passDesc.attachmentDescriptions.push_back(attachmentDesc);
178 pass.RenderPass = _graphicsDevice->CreateRenderPass(passDesc);
179 pass.RenderPassId = renderPassManager.RegisterRenderPass(pass.RenderPass);
180 });
181
182 if(_graphicsDevice == nullptr)
183 {
184 throw std::invalid_argument("A graphics device has not been provided. Call WithGraphicsDevice to provide one.");
185 }
186 else if( _resourceLoader == nullptr)
187 {
188 throw std::invalid_argument("A desktop resource loader has not been provided. Call WithResourceLoader to provide one.");
189 }
190 else if(_memoryAllocator == nullptr)
191 {
192 throw std::invalid_argument("A graphics memory allocator has not been provided. Call WithMemoryAllocator to provide one.");
193 }
194 else if (_context == nullptr)
195 {
196 throw std::invalid_argument("A graphics surface context has not been provided. Call WithSurfaceContext to provide one.");
197 }
198 else {
199 _defaultSpriteRenderer = std::make_shared<SpriteRendererSystem<TGraphicsBackend>>(
200 _graphicsDevice, pass, _resourceLoader, _memoryAllocator, _context);
201 }
202
203 return *this;
204 }
