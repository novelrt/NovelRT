35 inline VkBlendFactor GetVulkanBlendFactor(GraphicsPipelineBlendFactor blendFactor)
36 {
37 if (blendFactor >= GraphicsPipelineBlendFactor::__COUNT)
38 {
39 throw Exceptions::NotSupportedException(
40 "The specified blend factor is not supported on the default Metal pipeline.");
41 }
42 static_assert((sizeof(NovelRTToVulkanBlend) / sizeof(VkBlendFactor)) ==
43 static_cast<uint32_t>(GraphicsPipelineBlendFactor::__COUNT));
44 return NovelRTToVulkanBlend[static_cast<uint32_t>(blendFactor)];
45 }
