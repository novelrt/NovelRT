210  template<typename... TComponentTypes>
211  [[nodiscard]] SystemScheduler InitialiseAndRegisterComponents(
212  std::tuple<TComponentTypes, std::string>... deleteInstructionStatesAndSerialisedTypeNames)
213  {
214  SystemScheduler scheduler(_threadCount.value_or(0));
215 
216  if (_shouldAddDefaultSystems)
217  {
218  AddDefaultComponentsAndSystems(scheduler);
219  }
220 
221  for (auto&& system : _systems)
222  {
223  scheduler.RegisterSystem(system);
224  }
225 
226  for (auto&& iecsSystem : _iecsSystems)
227  {
228  scheduler.RegisterSystem(iecsSystem);
229  }
230 
231  scheduler.GetComponentCache().RegisterComponentType<TComponentTypes...>(
232  std::get<0>(deleteInstructionStatesAndSerialisedTypeNames)...,
233  std::get<1>(deleteInstructionStatesAndSerialisedTypeNames)...);
234  scheduler.SpinThreads();
235 
236  return scheduler;
237  }
