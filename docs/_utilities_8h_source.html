96  [[nodiscard]] inline constexpr int32_t PopCount(uint32_t value) noexcept
97  {
98 #if __cpp_lib_bitops
99  return std::popcount(value);
100 #else
101  const uint32_t c1 = 0x55555555u;
102  const uint32_t c2 = 0x33333333u;
103  const uint32_t c3 = 0x0F0F0F0Fu;
104  const uint32_t c4 = 0x01010101u;
105 
106  value -= (value >> 1) & c1;
107  value = (value & c2) + ((value >> 2) & c2);
108  value = (((value + (value >> 4)) & c3) * c4) >> 24;
109 
110  return static_cast<int32_t>(value);
111 #endif
112  }
