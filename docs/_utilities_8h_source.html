31  [[nodiscard]] inline static bool IsPow2(size_t value) noexcept
32  {
33  return ((value & (value - 1ULL)) == 0ULL) && (value != 0ULL);
34  }

36  [[nodiscard]] inline static size_t AlignUp(size_t address, size_t alignment)
37  {
38  if (!IsPow2(alignment))
39  {
40  throw Exceptions::InvalidOperationException("Alignment must be to a power of two.");
41  }
42 
43  return (address + (alignment - 1ULL)) & ~(alignment - 1ULL);
44  }

46  [[nodiscard]] [[deprecated("This implementation of PopCount is superseeded by std::popcount in C++20. This "
47  "will be removed when the language version is updated.")]] inline static int32_t
48  PopCount(uint32_t value) noexcept
49  {
50  const uint32_t c1 = 0x55555555u;
51  const uint32_t c2 = 0x33333333u;
52  const uint32_t c3 = 0x0F0F0F0Fu;
53  const uint32_t c4 = 0x01010101u;
54 
55  value -= (value >> 1) & c1;
56  value = (value & c2) + ((value >> 2) & c2);
57  value = (((value + (value >> 4)) & c3) * c4) >> 24;
58 
59  return static_cast<int32_t>(value);
60  }

62  [[nodiscard]] inline static uint32_t Log2(uint32_t value) noexcept
63  {
64  value |= value >> 1;
65  value |= value >> 2;
66  value |= value >> 4;
67  value |= value >> 8;
68  value |= value >> 16;
69 
70  return Log2DeBruijn()[(value * 0x07C4ACDCu) >> 27];
71  }

73  [[nodiscard]] inline static uint32_t LeadingZeroCount32(uint32_t value) noexcept
74  {
75  if (value == 0)
76  {
77  return 32;
78  }
79  else
80  {
81  return 31 ^ Log2(value);
82  }
83  }

85  [[nodiscard]] inline static uint64_t LeadingZeroCount64(uint64_t value) noexcept
86  {
87  uint32_t hi = static_cast<uint32_t>(value >> 32);
88 
89  if (hi == 0)
90  {
91  return 32 + LeadingZeroCount32(static_cast<uint32_t>(value));
92  }
93  else
94  {
95  return LeadingZeroCount32(hi);
