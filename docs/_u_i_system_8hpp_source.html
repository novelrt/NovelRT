65 void ParseElementCommands(Catalogue& catalogue, EntityId entity, Ecs::UI::Components::UIElement& element, float scaleX, float scaleY)
66 {
67 auto [containers, buttons, textView, transforms, clickEvents] =
68 catalogue.GetComponentViews<Ecs::UI::Components::UIWidgetContainer, Ecs::UI::Components::UIButton,
69 Ecs::UI::Components::UIText, Ecs::Components::TransformComponent,
70 Ecs::UI::Components::UIClickEvent>();
71
72 switch (element.Type)
73 {
74 case Ecs::UI::UIComponentType::Container:
75 {
76 GenerateContainerCommands(catalogue, entity, element, scaleX, scaleY);
77 break;
78 }
79 case Ecs::UI::UIComponentType::Button:
80 {
81 GenerateButtonCommands(catalogue, entity, element, scaleX, scaleY);
82 break;
83 }
84 case Ecs::UI::UIComponentType::Text:
85 {
86 GenerateTextLineCommands(catalogue, entity, element);
87 break;
88 }
89 default:
90 {
91 std::stringstream exceptionValue;
92 exceptionValue << "The UI Component type is undefined for the current element at entity " << entity << ".";
93 throw NovelRT::Exceptions::InvalidOperationException(exceptionValue.str());
94 }
95 }

98 void EnumerateChildren(EntityGraphView& graph,
99 ComponentView<Ecs::UI::Components::UIElement>& view,
100 std::vector<std::pair<EntityId, int32_t>>& inOrder,
101 int32_t currentDepth = 0)
102 {
103 if (view.HasComponent(graph.GetRawEntityId()))
104 inOrder.emplace_back(graph.GetRawEntityId(), currentDepth);
105
106 if (!graph.HasChildren())
107 return;
108
109 for (const auto& child : graph.GetOriginalChildren())
110 {
111 EnumerateChildren(child.get(), view, inOrder, currentDepth + 1);
112 }
113 }

203 void Render(Catalogue& catalogue)
204 {
205 auto [renderPasses, commandLists, trackedSemaphores] =
206 catalogue.GetComponentViews<Graphics::Components::RenderPass<TGraphicsBackend>,
207 Graphics::Components::BuiltCommandList<TGraphicsBackend>,
208 Graphics::Components::TrackedSemaphore<TGraphicsBackend>>();
209
210 // create upload command list
211 auto uiContext = _gfxDevice->CreateGraphicsContext();
212 auto uploadCmdList = uiContext->CreateCmdList();
213
214 // upload data to gpu
215 if (_uiProvider->UploadToGPU(uploadCmdList))
216 {
217 // if we have data to upload to the gpu, then submit the commands.
218 std::vector<std::shared_ptr<NovelRT::Graphics::GraphicsCmdList<TGraphicsBackend>>> lists{uploadCmdList};
219
220 // create the semaphore to signal when uploading
221 std::vector<
222 std::pair<std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>, uint64_t>>
223 uploadSemaphores{std::make_pair(_uploadSemaphore, ++_submittedUploads)};
224
225 _gfxDevice->QueueSubmit(lists, uploadSemaphores);
226
227 // Create a tracked semaphore that will wait on the upload semaphore
228 Graphics::Components::TrackedSemaphore<TGraphicsBackend> newUploadTracker{
229 .semaphore =
230 new std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>(_uploadSemaphore),
231 .signalValue = _submittedUploads};
232
233 // Create unique id for tracking upload
234 auto uploadId = catalogue.CreateEntity();
235
236 // add upload semaphore to tracked catalogue
237 trackedSemaphores.AddComponent(uploadId, newUploadTracker);
238
239 // create entity id for drawing
240 auto entityId = catalogue.CreateEntity();
241
242 // check if draw semaphore has value
243 // if not, create one
244 if (!_drawSemaphoreEntity.has_value())
245 {
246 _drawSemaphoreEntity = catalogue.CreateEntity();
247 }
248
249 // create the tracked (signal) semaphore component for the draw semaphore
250 Graphics::Components::TrackedSemaphore<TGraphicsBackend> signalSemaComponent{};
251 signalSemaComponent.isWaitSemaphore = false;
252 signalSemaComponent.semaphore =
253 new std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>(_drawSemaphore);
254 signalSemaComponent.signalValue = ++_frameCounter;
255
256 // add it to the catalogue
257 trackedSemaphores.AddComponent(_drawSemaphoreEntity.value(), signalSemaComponent);
258
259 // create the draw cmdList
260 auto drawCmdList =
261 uiContext->CreateCmdList(std::optional<NovelRT::Graphics::SecondaryCmdListInfo<TGraphicsBackend>>(
262 {_uiProvider->GetDedicatedRenderPass(), 0}));
263
264 drawCmdList->Begin();
265 // call render
266 _uiProvider->Draw(drawCmdList);
267 drawCmdList->End();
268 // Create renderpass component
269 Graphics::Components::RenderPass<TGraphicsBackend> passComponent{};
270 passComponent.renderPassIndex = _renderPassId;
271
272 // create draw lists
273 Graphics::Components::BuiltCommandList<TGraphicsBackend> drawLists{
274 .commandList =
275 new std::shared_ptr<NovelRT::Graphics::GraphicsCmdList<TGraphicsBackend>>(drawCmdList)};
276
277 // add to the catalogue
278 renderPasses.AddComponent(entityId, passComponent);
279 commandLists.AddComponent(entityId, drawLists);
280 }

283 void ProcessComponents(Timing::Timestamp delta, Catalogue& catalogue)
284 {
285 // 1. Start frame unanimously - this is for debug/metric windows
286 _uiProvider->BeginFrame(NovelRT::Timing::GetSeconds<float>(delta));
287
288 auto [graph, elementView, camView] =
289 catalogue.GetComponentViews<Ecs::Components::EntityGraphComponent, Ecs::UI::Components::UIElement, Graphics::Components::Camera>();
290
291 // 2. Determine the order in which we should enumerate the entities based on... something?
292 std::map<EntityId, Ecs::UI::Components::UIElement> elements{};
293 std::set<EntityId> rootEntities{};
294 std::vector<EntityGraphView> roots{};
295 std::vector<std::pair<EntityId, int32_t>> ordered{};
296
297 for (auto [entity, element] : elementView)
298 {
299 auto [iterator, inserted] = elements.try_emplace(entity);
300 iterator->second = element;
301
302 if (graph.HasComponent(entity))
303 {
304 auto root = GetRoot(graph, catalogue, entity);
305 auto [it, insertedTwo] = rootEntities.insert(root.GetRawEntityId());
306 if (insertedTwo)
307 roots.emplace_back(std::move(root));
308 }
309 }
310
311 if (elements.empty())
312 {
313 // end the frame now so at least debug stuff displays
314 _uiProvider->EndFrame();
315 return;
316 }
317
318 // 3. Enumerate the child elements
319 for (auto& root : roots)
320 {
321 EnumerateChildren(root, elementView, ordered);
322 }
323
324 // 4a - SCALE EVERYTHING
325 // This lets us make sure that imgui tracks with the screen properly - otherwise it does not adjust
326 // with other elements.
327 auto& io = ::ImGui::GetIO();
328
329 float refWidth = io.DisplaySize.x;
330 float refHeight = io.DisplaySize.y;
331
332 for (auto [entity, camera]: camView)
333 {
334 if (camera.isScreenSpace)
335 {
336 refWidth = static_cast<float>(camera.referenceResolutionWidth);
337 refHeight = static_cast<float>(camera.referenceResolutionHeight);
338 break;
339 }
340 }
341
342 float scaleX = io.DisplaySize.x / refWidth;
343 float scaleY = io.DisplaySize.y / refHeight;
344 //float avgScale = (scaleX + scaleY) * 0.5f;
345 //io.FontGlobalScale *= avgScale; // font scaling
346
347 // 4b. write Imgui commands
348 std::vector<std::pair<EntityId, int32_t>> containers{};
349
350 for (auto& [entity, depth] : ordered)
351 {
352 while (!containers.empty() && containers.back().second >= depth)
353 {
354 ImGui::End();
355 containers.pop_back();
356 }
357
358 auto element = elements.find(entity);
359
360 if (element != elements.end())
361 {
362 ParseElementCommands(catalogue, entity, element->second, scaleX, scaleY);
363
364 if (element->second.Type == UIComponentType::Container)
365 {
366 containers.emplace_back(entity, depth);
367 }
368 }
369 }
370
371 while (!containers.empty())
372 {
373 ImGui::End();
374 containers.pop_back();
375 }
376
377 // 5. End imgui frame so all items are writen to draw data
378 _uiProvider->EndFrame();
379 }
