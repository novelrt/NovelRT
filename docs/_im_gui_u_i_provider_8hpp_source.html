119 inline void CreateDedicatedRenderPass()
120 {
121 // Create Render Pass
122 if (_renderpass == nullptr)
123 {
124 NovelRT::Graphics::GraphicsRenderPassDescription passDesc{};
125 NovelRT::Graphics::GraphicsAttachmentDescription attachmentDesc{};
126
127 attachmentDesc.texelFormat = _graphicsDevice->GetSwapchain()->GetFormat();
128
129 attachmentDesc.loadOp = NovelRT::Graphics::LoadOp::Load;
130 attachmentDesc.storeOp = NovelRT::Graphics::StoreOp::Store;
131 attachmentDesc.initialLayout = NovelRT::Graphics::ImageLayout::Present;
132 attachmentDesc.finalLayout = NovelRT::Graphics::ImageLayout::Present;
133
134 passDesc.attachmentDescriptions.push_back(attachmentDesc);
135 _renderpass = _graphicsDevice->CreateRenderPass(passDesc);
136 }
137
138 // Load shader files for ImGui
139 auto vertImguiShader = _resourceLoader->LoadShaderSource("imgui_vert.spv");
140 auto pixelImguiShader = _resourceLoader->LoadShaderSource("imgui_frag.spv");
141
142 // Define elements that make up vertex shader
143 std::vector<GraphicsPipelineInputElement> elements{
144 GraphicsPipelineInputElement(typeid(NovelRT::Maths::GeoVector2F),
145 GraphicsPipelineInputElementKind::Position, 8),
146 GraphicsPipelineInputElement(typeid(NovelRT::Maths::GeoVector2F),
147 GraphicsPipelineInputElementKind::Normal, 8),
148 GraphicsPipelineInputElement(typeid(ImU32), GraphicsPipelineInputElementKind::Colour, sizeof(ImU32))};
149
150 // Because we use push constants, define them here
151 std::vector<GraphicsPushConstantRange> pushConstants{
152 GraphicsPushConstantRange{ShaderProgramVisibility::Vertex, 0, sizeof(float) * 4}};
153
154 std::vector<GraphicsPipelineInput> inputs{GraphicsPipelineInput(elements)};
155 std::vector<GraphicsPipelineResource> resources{
156 GraphicsPipelineResource(GraphicsPipelineResourceKind::Texture, ShaderProgramVisibility::Pixel)};
157 // Create pipeline signature
158 _pipelineSignature = _graphicsDevice->CreatePipelineSignature(GraphicsPipelineBlendFactor::SrcAlpha,
159 GraphicsPipelineBlendFactor::OneMinusSrcAlpha,
160 inputs, resources, pushConstants);
161 auto imVertProg =
162 _graphicsDevice->CreateShaderProgram("main", ShaderProgramKind::Vertex, vertImguiShader.shaderCode);
163 auto imPixProg =
164 _graphicsDevice->CreateShaderProgram("main", ShaderProgramKind::Pixel, pixelImguiShader.shaderCode);
165
166 // Create pipeline
167 _pipeline = _graphicsDevice->CreatePipeline(_pipelineSignature, imVertProg, imPixProg, _renderpass, true);
168 }

269 void BeginFrame(float deltaTime)
270 {
271 auto& io = ::ImGui::GetIO();
272 auto windowSize = _windowProvider->GetSize();
273 auto windowScale = _windowProvider->GetWindowScale();
274 io.DisplaySize = ImVec2(windowSize.x, windowSize.y);
275 io.DisplayFramebufferScale = ImVec2(windowScale.x, windowScale.y);
276 io.DeltaTime = deltaTime;
277
278 float scaleX = windowSize.x / _initialWindowSize.x;
279 float scaleY = windowSize.y / _initialWindowSize.y;
280
281 io.FontGlobalScale = (scaleX + scaleY) / 2.0f;
282
283 auto mousePos = _inputProvider->GetRawMousePosition();
284 io.AddMousePosEvent(mousePos.x, mousePos.y);
285 auto state = _inputProvider->GetKeyState(_mouseInputAction.actionName);
286 if (state == KeyState::KeyDown || state == KeyState::KeyDownHeld)
287 {
288 io.AddMouseButtonEvent(0, true);
289 }
290 else
291 {
292 io.AddMouseButtonEvent(0, false);
293 }
294
295 ::ImGui::NewFrame();
296 if (_showMetricsWindow)
297 {
298 ::ImGui::ShowMetricsWindow(&_showMetricsWindow);
299 }
300 }

303 {
304 ::ImGui::Render();
305
306 for (auto it = _bufferCache.begin(); it != _bufferCache.end();)
307 {
308 it->frameLifetime--;
309 if (it->frameLifetime <= 0)
310 {
311 it = _bufferCache.erase(it);
312 }
313 else
314 {
315 it++;
316 }
317 }
318
319 for (auto it = _cmdListCache.begin(); it != _cmdListCache.end();)
320 {
321 it->frameLifetime--;
322 if (it->frameLifetime <= 0)
323 {
324 it = _cmdListCache.erase(it);
325 }
326 else
327 {
328 it++;
329 }
330 }
331
332 for (auto it = _descriptorSetCache.begin(); it != _descriptorSetCache.end();)
333 {
334 it->frameLifetime--;
335 if (it->frameLifetime <= 0)
336 {
337 it = _descriptorSetCache.erase(it);
338 }
339 else
340 {
341 it++;
342 }

346 bool UploadToGPU(std::shared_ptr<Graphics::GraphicsCmdList<TGraphicsBackend>> currentCmdList)
347 {
348 ImDrawData* drawData = ::ImGui::GetDrawData();
349
350 if (drawData == nullptr || drawData->TotalVtxCount <= 0)
351 {
352 _drawEnabled = false;
353 return false;
354 }
355 _drawEnabled = true;
356 currentCmdList->Begin();
357
358 size_t vertexSize = static_cast<size_t>(drawData->TotalVtxCount) * sizeof(ImDrawVert);
359 size_t indexSize = static_cast<size_t>(drawData->TotalIdxCount) * sizeof(ImDrawIdx);
360
361 // Create buffers
362 GraphicsBufferCreateInfo bufferCreateInfo{};
363 bufferCreateInfo.cpuAccessKind = GraphicsResourceAccess::Write;
364 bufferCreateInfo.gpuAccessKind = GraphicsResourceAccess::Read;
365 bufferCreateInfo.size = vertexSize;
366 auto vertexStagingBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
367 _bufferCache.emplace_back(CachedBufferObject{vertexStagingBuffer, _defaultFramePersistenceCount});
368
369 bufferCreateInfo.bufferKind = GraphicsBufferKind::Vertex;
370 bufferCreateInfo.cpuAccessKind = GraphicsResourceAccess::None;
371 bufferCreateInfo.gpuAccessKind = GraphicsResourceAccess::Write;
372 auto vertexBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
373 _bufferCache.emplace_back(CachedBufferObject{vertexBuffer, _defaultFramePersistenceCount});
374
375 // Create index buffer + staging
376 bufferCreateInfo.bufferKind = GraphicsBufferKind::Default;
377 bufferCreateInfo.cpuAccessKind = GraphicsResourceAccess::Write;
378 bufferCreateInfo.gpuAccessKind = GraphicsResourceAccess::Read;
379 bufferCreateInfo.size = indexSize;
380 auto indexStagingBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
381 _bufferCache.emplace_back(CachedBufferObject{indexStagingBuffer, _defaultFramePersistenceCount});
382
383 bufferCreateInfo.bufferKind = GraphicsBufferKind::Index;
384 bufferCreateInfo.cpuAccessKind = GraphicsResourceAccess::None;
385 bufferCreateInfo.gpuAccessKind = GraphicsResourceAccess::Write;
386 auto indexBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
387 _bufferCache.emplace_back(CachedBufferObject{indexBuffer, _defaultFramePersistenceCount});
388
389 // Allocate buffers
390 auto vertexBufferRegion = vertexBuffer->Allocate(vertexSize, 4);
391 auto vertexStageBufferRegion = vertexStagingBuffer->Allocate(vertexSize, 4);
392 auto indexBufferRegion = indexBuffer->Allocate(indexSize, 4);
393 auto indexStageBufferRegion = indexStagingBuffer->Allocate(indexSize, 4);
394
395 // Map vertex buffer to region
396 auto pVertexBuffer = vertexStagingBuffer->template Map<ImDrawVert>(vertexBufferRegion);
397 auto pVertPtr = pVertexBuffer.data();
398 // Map index buffer to region
399 auto pIndexbuffer = indexStagingBuffer->template Map<ImDrawIdx>(indexBufferRegion);
400 auto pIdxPtr = pIndexbuffer.data();
401
402 // Slamjam the vertex buffer and index buffer data into their regions
403 for (int i = 0; i < drawData->CmdListsCount; i++)
404 {
405 ImDrawList* list = drawData->CmdLists[i];
406
407 // Copy vertex and index data
408 memcpy(pVertPtr, list->VtxBuffer.Data, static_cast<size_t>(list->VtxBuffer.Size) * sizeof(ImDrawVert));
409 memcpy(pIdxPtr, list->IdxBuffer.Data, static_cast<size_t>(list->IdxBuffer.Size) * sizeof(ImDrawIdx));
410
411 pVertPtr += list->VtxBuffer.Size;
412 pIdxPtr += list->IdxBuffer.Size;
413 }
414
415 // Unmap and copy to GPU
416 vertexStagingBuffer->UnmapAndWrite(vertexBufferRegion);
417 indexStagingBuffer->UnmapAndWrite(indexBufferRegion);
418 currentCmdList->CmdCopy(vertexBufferRegion, vertexStageBufferRegion);
419 currentCmdList->CmdCopy(indexBufferRegion, indexStageBufferRegion);
420
421 // Sync the commands so that we can prevent data issues
422 currentCmdList->CmdPipelineBufferBarrier(
423 vertexBuffer, GraphicsMemoryAccessMode::TransferWrite, GraphicsMemoryAccessMode::VertexAttributeRead,
424 GraphicsPipelineVisibility::Transfer, GraphicsPipelineVisibility::VertexInput);
425 currentCmdList->CmdPipelineBufferBarrier(
426 indexBuffer, GraphicsMemoryAccessMode::TransferWrite, GraphicsMemoryAccessMode::IndexRead,
427 GraphicsPipelineVisibility::Transfer, GraphicsPipelineVisibility::VertexInput);
428
429 _currentIndexBufferRegion = indexBufferRegion;
430 _currentVertexBuffer = vertexBuffer;
431 _currentVertexBufferRegion = vertexBufferRegion;
432 _cachedDrawData = drawData;
433 currentCmdList->End();
434
435 _cmdListCache.emplace_back(CachedCmdListObject{currentCmdList, _defaultFramePersistenceCount});
436
437 return true;
438 }

440 void Draw(std::shared_ptr<Graphics::GraphicsCmdList<TGraphicsBackend>> currentCmdList)
441 {
442 if (!_drawEnabled)
443 return;
444
445 auto drawData = _cachedDrawData;
446
447 auto graphicsSurface = _graphicsDevice->GetSurface();
448 float surfaceWidth = graphicsSurface->GetWidth();
449 float surfaceHeight = graphicsSurface->GetHeight();
450
451 int32_t globalVertexOffset = 0;
452 int32_t globalIndexOffset = 0;
453
454 ImVec2 clippingOffset = drawData->DisplayPos;
455 // ImVec2 clippingScale = drawData->FramebufferScale;
456
457 // Bind the Vertex Buffers and the index buffer region
458 std::array<std::reference_wrapper<const std::shared_ptr<GraphicsBuffer<TGraphicsBackend>>>, 1> buffers{
459 std::cref(_currentVertexBuffer)};
460 auto currentOffset = _currentVertexBufferRegion->GetOffset();
461 auto currentIndexBufferRegion = _currentIndexBufferRegion;
462 std::array<size_t, 1> offsets{currentOffset};
463
464 ViewportInfo viewportInfoStruct{};
465 viewportInfoStruct.x = 0;
466 viewportInfoStruct.y = 0;
467 viewportInfoStruct.width = surfaceWidth;
468 viewportInfoStruct.height = surfaceHeight;
469 viewportInfoStruct.minDepth = 0.0f;
470 viewportInfoStruct.maxDepth = 1.0f;
471
472 currentCmdList->CmdSetViewport(viewportInfoStruct);
473
474 currentCmdList->CmdBindPipeline(_pipeline);
475 currentCmdList->CmdBindVertexBuffers(0, 1, buffers, offsets);
476 currentCmdList->CmdBindIndexBuffer(_currentIndexBufferRegion, IndexType::UInt16);
477
478 float scale[2];
479 scale[0] = 2.0f / drawData->DisplaySize.x;
480 scale[1] = 2.0f / drawData->DisplaySize.y;
481 float translate[2];
482 translate[0] = -1.0f - drawData->DisplayPos.x * scale[0];
483 translate[1] = -1.0f - drawData->DisplayPos.y * scale[1];
484 NovelRT::Utilities::Span<float> scaleSpan(scale);
485 NovelRT::Utilities::Span<float> translateSpan(translate);
486
487 currentCmdList->CmdPushConstants(_pipelineSignature, ShaderProgramVisibility::Vertex, 0,
488 NovelRT::Utilities::SpanCast<uint8_t>(scaleSpan));
489 currentCmdList->CmdPushConstants(_pipelineSignature, ShaderProgramVisibility::Vertex, sizeof(float) * 2,
490 NovelRT::Utilities::SpanCast<uint8_t>(translateSpan));
491
492 currentCmdList->CmdSetScissor(
493 NovelRT::Maths::GeoVector2F::Zero(),
494 NovelRT::Maths::GeoVector2F(drawData->DisplaySize.x, drawData->DisplaySize.y));
495
496 for (int n = 0; n < drawData->CmdListsCount; n++)
497 {
498 const ImDrawList* list = drawData->CmdLists[n];
499 for (int cmdIndex = 0; cmdIndex < list->CmdBuffer.Size; cmdIndex++)
500 {
501 const ImDrawCmd* drawCommand = &list->CmdBuffer[cmdIndex];
502 if (drawCommand->UserCallback != nullptr)
503 {
504 drawCommand->UserCallback(list, drawCommand);
505 }
506 else
507 {
508
509 // Project scissor/clipping rectangles into framebuffer space
510 ImVec2 clippingMin((drawCommand->ClipRect.x - clippingOffset.x) * 1, //* clippingScale.x,
511 (drawCommand->ClipRect.y - clippingOffset.y) * 1); //* clippingScale.y);
512 ImVec2 clippingMax((drawCommand->ClipRect.z - clippingOffset.x) * 1, //* clippingScale.x,
513 (drawCommand->ClipRect.w - clippingOffset.y) * 1); //* clippingScale.y);
514
515 // Clamp to viewport as vkCmdSetScissor() won't accept values that are off bounds
516 if (clippingMin.x < 0.0f)
517 {
518 clippingMin.x = 0.0f;
519 }
520 if (clippingMin.y < 0.0f)
521 {
522 clippingMin.y = 0.0f;
523 }
524 if (clippingMax.x > surfaceWidth)
525 {
526 clippingMax.x = surfaceWidth;
527 }
528 if (clippingMax.y > surfaceHeight)
529 {
530 clippingMax.y = (float)surfaceHeight;
531 }
532 if (clippingMax.x <= clippingMin.x || clippingMax.y <= clippingMin.y)
533 continue;
534
535 // Apply scissor/clipping rectangle
536 currentCmdList->CmdSetScissor(NovelRT::Maths::GeoVector2F(clippingMin.x, clippingMin.y),
537 NovelRT::Maths::GeoVector2F((clippingMax.x - clippingMin.x),
538 (clippingMax.y - clippingMin.y)));
539
540 // Bind DescriptorSet with font or user texture
541
542 auto it = _textureMap.find(drawCommand->GetTexID());
543 if (it == _textureMap.end())
544 continue; // TODO: this should probably warn
545
546 std::vector<std::shared_ptr<GraphicsResourceMemoryRegion<GraphicsResource, TGraphicsBackend>>>
547 inputResourceRegions{it->second.region};
548
549 // Specify the descriptor set data
550 auto descriptorSetData = _pipeline->CreateDescriptorSet();
551 descriptorSetData->AddMemoryRegionsToInputs(inputResourceRegions);
552 descriptorSetData->UpdateDescriptorSetData();
553
554 _descriptorSetCache.emplace_back(
555 CachedDescriptorSetObject{descriptorSetData, _defaultFramePersistenceCount});
556
557 // Bind the descriptor set
558 std::array<
559 std::reference_wrapper<const std::shared_ptr<GraphicsDescriptorSet<TGraphicsBackend>>>, 1>
560 descriptorData{std::cref(descriptorSetData)};
561
562 currentCmdList->CmdBindDescriptorSets(descriptorData);
563
564 currentCmdList->CmdDrawIndexed(
565 drawCommand->ElemCount, 1,
566 drawCommand->IdxOffset + static_cast<uint32_t>(globalIndexOffset),
567 drawCommand->VtxOffset + static_cast<uint32_t>(globalVertexOffset), 0);
568 }
569 }
570
571 globalIndexOffset += list->IdxBuffer.Size;
572 globalVertexOffset += list->VtxBuffer.Size;
573 }
574
575 // Reset clipping rect as per imgui
576 currentCmdList->CmdSetScissor(
577 NovelRT::Maths::GeoVector2F::Zero(),
578 NovelRT::Maths::GeoVector2F(drawData->DisplaySize.x, drawData->DisplaySize.y));
579 }

586 inline void UploadFontData(std::shared_ptr<Graphics::GraphicsCmdList<TGraphicsBackend>> cmdList)
587 {
588 uint8_t* pixels;
589 int32_t width;
590 int32_t height;
591 auto& io = ::ImGui::GetIO();
592 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
593
594 // Create the texture
595 auto textureCreateInfo = GraphicsTextureCreateInfo{GraphicsTextureAddressMode::Repeat,
596 GraphicsTextureKind::TwoDimensional,
597 GraphicsResourceAccess::None,
598 GraphicsResourceAccess::ReadWrite,
599 static_cast<uint32_t>(width),
600 static_cast<uint32_t>(height),
601 1,
602 GraphicsMemoryRegionAllocationFlags::None,
603 TexelFormat::R8G8B8A8_UNORM};
604 auto texture2D = _memoryAllocator->CreateTexture(textureCreateInfo);
605
606 GraphicsBufferCreateInfo bufferCreateInfo{};
607 bufferCreateInfo.cpuAccessKind = GraphicsResourceAccess::Write;
608 bufferCreateInfo.gpuAccessKind = GraphicsResourceAccess::Read;
609 bufferCreateInfo.size = texture2D->GetSize();
610
611 // Create Texture Staging Buffer
612 auto textureStagingBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
613
614 // Begin commands
615 // context->BeginFrame();
616
617 // Allocate region based on size of texture
618 auto texture2DRegion = texture2D->Allocate(texture2D->GetSize(), 4);
619
620 // Create a staging buffer region for texture
621 auto stagingBufferRegion = textureStagingBuffer->Allocate(texture2D->GetSize(), 4);
622 auto textureData = textureStagingBuffer->template Map<uint8_t>(stagingBufferRegion);
623
624 // Write the data over before copying
625 auto pixelSpan = NovelRT::Utilities::Span<const uint8_t>(pixels, static_cast<size_t>(width * height * 4));
626 // memcpy(textureData.data(), pixels, (width * height) * sizeof(char) * 4);
627 NovelRT::Utilities::Memory::Copy(pixelSpan, textureData);
628 textureStagingBuffer->UnmapAndWrite(stagingBufferRegion);
629
630 // Set texture ID so that ImGui can refer back to proper id during render
631 io.Fonts->SetTexID(0);
632
633 // Send the data to GPU
634 cmdList->Begin();
635 cmdList->CmdBeginTexturePipelineBarrierLegacyVersion(texture2D);
636 cmdList->CmdCopy(texture2D, stagingBufferRegion);
637 cmdList->CmdEndTexturePipelineBarrierLegacyVersion(texture2D);
638 cmdList->End();
639
