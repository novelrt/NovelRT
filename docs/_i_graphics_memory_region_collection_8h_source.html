286  [[nodiscard]] bool TryAllocate(size_t size, size_t alignment, GraphicsMemoryRegion<TSelf>& outRegion) final
287  {
288  if (size == 0)
289  {
290  throw std::out_of_range(
291  "The size parameter was set to 0. This is not a valid size and is out of range.");
292  }
293 
294  if (!Maths::Utilities::IsPow2(alignment))
295  {
296  throw std::out_of_range("The provided alignment is invalid, as it is not to a power of two.");
297  }
298 
299  size_t sizeWithMargins = size + (2 * GetMinimumAllocatedRegionMarginSize());
300  bool allocatedRegion = false;
301 
302  if (GetTotalFreeRegionSize() >= sizeWithMargins)
303  {
304  gsl::span<typename std::list<GraphicsMemoryRegion<TSelf>>::iterator> freeRegionsBySizeSpan(
305  *_freeRegionsBySize);
306  size_t freeRegionsBySizeLength = freeRegionsBySizeSpan.size();
307 
308  if (freeRegionsBySizeLength > 0)
309  {
310  for (auto index = BinarySearchFirstRegionNodeWithSizeNotLessThan(sizeWithMargins);
311  index < freeRegionsBySizeLength; ++index)
312  {
313  auto regionNode = freeRegionsBySizeSpan[index];
314 
315  if (TryAllocateInternal(size, alignment, regionNode))
316  {
317  outRegion = *regionNode;
318  allocatedRegion = true;
319  break;
320  }
321  }
322  }
323  }
324 
325  if (!allocatedRegion)
326  {
327  outRegion = GraphicsMemoryRegion<TSelf>();
328  }
329 
330  assert(Validate());
331 
332  return allocatedRegion;
333  }

348  [[nodiscard]] bool Validate()
349  {
350  if (_regions->size() == 0)
351  {
352  return false;
353  }
354 
355  size_t calculatedSize = 0ULL;
356  size_t calculatedTotalFreeRegionSize = 0ULL;
357 
358  int32_t calculatedFreeRegionCount = 0;
359  size_t calculatedFreeRegionsToRegisterCount = 0;
360 
361  bool isPreviousRegionFree = false;
362 
363  for (auto&& region : *_regions)
364  {
365  if (region.GetOffset() != calculatedSize)
366  {
367  return false;
368  }
369 
370  bool isCurrentRegionFree = !region.GetIsAllocated();
371 
372  if (isPreviousRegionFree && isCurrentRegionFree)
373  {
374  return false;
375  }
376 
377  if (isCurrentRegionFree)
378  {
379  calculatedTotalFreeRegionSize += region.GetSize();
380  ++calculatedFreeRegionCount;
381 
382  if (region.GetSize() >= GetMinimumFreeRegionSizeToRegister())
383  {
384  ++calculatedFreeRegionsToRegisterCount;
385  }
386 
387  if (region.GetSize() < GetMinimumAllocatedRegionMarginSize())
388  {
389  return false;
390  }
391  }
392  else if (GetMinimumAllocatedRegionMarginSize() != 0 && !isPreviousRegionFree)
393  {
394  return false;
395  }
396 
397  calculatedSize += region.GetSize();
398  isPreviousRegionFree = isCurrentRegionFree;
399  }
400 
401  if (!ValidateFreeRegionsBySizeList())
402  {
403  return false;
404  }
405 
406  return calculatedSize == GetSize() && calculatedTotalFreeRegionSize == _totalFreeRegionSize &&
407  calculatedFreeRegionCount == _freeRegionCount &&
408  calculatedFreeRegionsToRegisterCount == _freeRegionsBySize->size();
409  }

412  [[nodiscard]] bool ValidateFreeRegionsBySizeList()
413  {
414  size_t lastRegionSize = 0ULL;
415 
416  for (auto&& region : *_freeRegionsBySize)
417  {
418  if (region->GetIsAllocated() || region->GetSize() < GetMinimumFreeRegionSizeToRegister() ||
419  region->GetSize() < lastRegionSize)
420  {
421  return false;
422  }
423 
424  lastRegionSize = region->GetSize();
425  }
426 
427  return true;

430  [[nodiscard]] size_t BinarySearchFirstRegionNodeWithSizeNotLessThan(size_t size) const noexcept
431  {
432  gsl::span<const typename std::list<GraphicsMemoryRegion<TSelf>>::iterator> freeRegionsBySizeSpan(
433  &(*_freeRegionsBySize->begin()), _freeRegionsBySize->size());
434 
435  size_t index = 0;
436  size_t endIndex = freeRegionsBySizeSpan.size();
437 
438  while (index < endIndex)
439  {
440  size_t midIndex = (index + endIndex) / 2;
441 
442  if (freeRegionsBySizeSpan[midIndex]->GetSize() < size)
443  {
444  index = midIndex + 1;
445  }
446  else
447  {
448  endIndex = midIndex;
449  }
450  }
451 
452  return index;
453  }

455  [[nodiscard]] typename std::list<GraphicsMemoryRegion<TSelf>>::iterator FreeRegionInternal(
456  typename std::list<GraphicsMemoryRegion<TSelf>>::iterator regionNode)
457  {
458  GraphicsMemoryRegion<TSelf>& region = *regionNode;
459 
460  if (!region.GetIsAllocated())
461  {
462  return regionNode;
463  }
464 
465  region = GraphicsMemoryRegion<TSelf>(region.GetAlignment(), region.GetCollection(), region.GetDevice(),
466  false, region.GetOffset(), region.GetSize());
467 
468  ++_freeRegionCount;
469  _totalFreeRegionSize += region.GetSize();
470 
471  auto nextRegionNode = regionNode;
472  std::advance(nextRegionNode, 1);
473  bool mergeWithNext = nextRegionNode != _regions->end() && !nextRegionNode->GetIsAllocated();
474 
475  auto previousRegionNode = regionNode;
476  std::advance(previousRegionNode, 1);
477  bool mergeWithPrevious = previousRegionNode != _regions->end() && !previousRegionNode->GetIsAllocated();
478 
479  if (mergeWithNext)
480  {
481  UnregisterFreeRegion(nextRegionNode);
482  MergeFreeRegionWithNext(regionNode);
483  }
484 
485  if (mergeWithPrevious)
486  {
487  UnregisterFreeRegion(previousRegionNode);
488  MergeFreeRegionWithNext(previousRegionNode);
489  RegisterFreeRegion(previousRegionNode);
490  return previousRegionNode;
491  }
492  else
493  {
494  RegisterFreeRegion(regionNode);
495  return regionNode;
496  }

499  void MergeFreeRegionWithNext(typename std::list<GraphicsMemoryRegion<TSelf>>::iterator regionNode)
500  {
501  auto nextRegionNode = regionNode;
502  ++nextRegionNode;
503 
504  if (nextRegionNode->GetIsAllocated())
505  {
506  throw Exceptions::InvalidOperationException(
507  "An allocated memory region was designated to be merged with an unallocated one.");
508  }
509

511  const GraphicsMemoryRegion<TSelf>& nextRegion = *nextRegionNode;
512 
513  region = GraphicsMemoryRegion<TSelf>(region.GetAlignment(), region.GetCollection(), region.GetDevice(),
514  region.GetIsAllocated(), region.GetOffset(),
515  region.GetSize() + nextRegion.GetSize());
516 
517  --_freeRegionCount;
518

522  void RegisterFreeRegion(typename std::list<GraphicsMemoryRegion<TSelf>>::iterator regionNode)
523  {
524  if (regionNode->GetIsAllocated())
525  {
526  throw Exceptions::InvalidOperationException(
527  "An allocated memory region was designated to be registered as free.");
528  }
529 
530  if (regionNode->GetSize() == 0)
531  {
532  throw Exceptions::InvalidOperationException(
533  "A memory region of size 0 was provided as a free region to register.");
534  }
535 
536  assert(ValidateFreeRegionsBySizeList());
537 
538  if (regionNode->GetSize() >= GetMinimumFreeRegionSizeToRegister())
539  {
540  if (_freeRegionsBySize->size() == 0)
541  {
542  _freeRegionsBySize->emplace_back(regionNode);
543  }
544  else
545  {

547  _freeRegionsBySize->insert(_freeRegionsBySize->begin() + index, regionNode);

554  [[nodiscard]] bool TryAllocateInternal(size_t size,
555  size_t alignment,
556  typename std::list<GraphicsMemoryRegion<TSelf>>::iterator regionNode)
557  {
558  if (size <= 0)
559  {
560  throw Exceptions::InvalidOperationException(
561  "An attempt to allocate a memory region of size 0 (or less) was made.");
562  }
563 
564  if (regionNode == _collection->end())
565  {
566  throw std::out_of_range("Parameter name: regionNode");
567  }
568 
569  GraphicsMemoryRegion<TSelf>& region = *regionNode;
570 
571  if (region.GetIsAllocated())
572  {
573  throw Exceptions::InvalidOperationException(
574  "An attempt to allocate a memory region that was already allocated was made.");
575  }
576 
577  if (region.GetSize() < size)
578  {
579  return false;
580  }
581 
582  size_t offset = region.GetOffset();
583 
584  if (GetMinimumAllocatedRegionMarginSize() > 0)
585  {
586  offset += GetMinimumAllocatedRegionMarginSize();
587  }
588 
589  offset = Maths::Utilities::AlignUp(offset, alignment);
590  size_t paddingBegin = offset - region.GetOffset();
591  size_t requiredEndMargin = GetMinimumAllocatedRegionMarginSize();
592 
593  if ((paddingBegin + size + requiredEndMargin) > size)
594  {
595  return false;
596  }
597 
598  size_t paddingEnd = region.GetSize() - paddingBegin - size;
599 
600  UnregisterFreeRegion(regionNode);
601 
602  region = GraphicsMemoryRegion<TSelf>(alignment, _collection, GetDevice(), true, offset, size);
603 
604  if (paddingEnd != 0)
605  {
606  auto iterator = _regions->insert(
607  std::next(regionNode),
608  GraphicsMemoryRegion<TSelf>(1, _collection, GetDevice(), false, offset + size, paddingEnd));
609  RegisterFreeRegion(iterator);
610  }
611 
612  if (paddingBegin != 0)
613  {
614  auto iterator =
615  _regions->insert(regionNode, GraphicsMemoryRegion<TSelf>(1, _collection, GetDevice(), false,
616  offset - paddingBegin, paddingBegin));
617  RegisterFreeRegion(iterator);
618  }
619 
620  --_freeRegionCount;
621 
622  if (paddingBegin > 0)
623  {
624  ++_freeRegionCount;
625  }
626 
627  if (paddingEnd > 0)
628  {
629  ++_freeRegionCount;
630  }
631 
632  _totalFreeRegionSize -= size;
633  return true;
634  }

636  void UnregisterFreeRegion(typename std::list<GraphicsMemoryRegion<TSelf>>::iterator regionNode)
637  {
638  if (regionNode->GetIsAllocated())
639  {
640  throw Exceptions::InvalidOperationException("An attempt was made to unregister a free region, but "
641  "the target region is currently allocated.");
642  }
643 
644  if (regionNode->GetSize() <= 0)
645  {
646  throw Exceptions::InvalidOperationException(
647  "An attempt was made to unregister an invalid region that has a size of 0 (or less).");
648  }
649 
650  assert(ValidateFreeRegionsBySizeList());
651 
652  if (regionNode->GetSize() >= GetMinimumFreeRegionSizeToRegister())
653  {
654  for (auto index = BinarySearchFirstRegionNodeWithSizeNotLessThan(regionNode->GetSize());
655  index < _freeRegionsBySize->size(); ++index)
656  {

662

664  {

669  }
