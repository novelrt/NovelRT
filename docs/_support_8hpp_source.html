14 [[nodiscard]] inline QueueFamilyIndices FindQueueFamilies(VkPhysicalDevice physicalDevice,
15 VkSurfaceKHR surface) noexcept
16 {
17 QueueFamilyIndices returnObject{};
18
19 uint32_t queueFamilyCount = 0;
20 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, nullptr);
21
22 std::vector<VkQueueFamilyProperties> queueFamilies(queueFamilyCount);
23 vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueFamilyCount, queueFamilies.data());
24
25 VkBool32 presentSupport = false;
26
27 uint32_t familyIndex = 0;
28 for (const auto& queueFamily : queueFamilies)
29 {
30 if (!returnObject.graphicsFamily.has_value() && (queueFamily.queueFlags & VK_QUEUE_GRAPHICS_BIT))
31 {
32 returnObject.graphicsFamily = familyIndex;
33 }
34
35 if (queueFamily.queueFlags & VK_QUEUE_TRANSFER_BIT)
36 {
37 if (!returnObject.transferFamily.has_value() ||
38 (Maths::Utilities::PopCount(queueFamily.queueFlags) <=
39 Maths::Utilities::PopCount(returnObject.transferFamily.value())))
40 {
41 returnObject.transferFamily = familyIndex;
42 }
43 }
44
45 vkGetPhysicalDeviceSurfaceSupportKHR(physicalDevice, familyIndex, surface, &presentSupport);
46 if (!returnObject.presentFamily.has_value() && presentSupport)
47 {
48 returnObject.presentFamily = familyIndex;
49 }
50
51 if (returnObject.IsComplete())
52 {
53 break;
54 }
55
56 ++familyIndex;
57 }
58
59 return returnObject;
60 }
