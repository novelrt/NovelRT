140 void HandleNewSpriteTextures(Catalogue catalogue)
141 {
142 auto lastCompletedUpload = _uploadSemaphore->GetValue();
143 if (lastCompletedUpload > 0)
144 {
145 lastCompletedUpload -= 1;
146 }
147
148 _pendingStagingBuffers.erase(std::remove_if(_pendingStagingBuffers.begin(), _pendingStagingBuffers.end(),
149 [lastCompletedUpload](const auto& pending)
150 { return pending.uploadIndex <= lastCompletedUpload; }),
151 _pendingStagingBuffers.end());
152
153 auto spriteView = catalogue.GetComponentView<Components::Sprite>();
154
155 for (auto [entityId, sprite] : spriteView)
156 {
157 if (_textureCache.find(sprite.assetId) != _textureCache.end() ||
158 _pendingLoads.find(sprite.assetId) != _pendingLoads.end())
159 {
160 continue;
161 }
162
163 _pendingLoads.insert(sprite.assetId);
164
165 catalogue.ScheduleWithCompletion(
166 [this, assetId = sprite.assetId]() { return _resourceLoader->LoadTexture(assetId); },
167 [this, assetId = sprite.assetId](Timing::Timestamp /*delta*/, Catalogue completionCatalogue,
168 ResourceManagement::TextureMetadata textureMetadata)
169 {
170
171 auto texture2D = _memoryAllocator->CreateTexture2DRepeatGpuWriteOnly(textureMetadata.width,
172 textureMetadata.height);
173
174 NovelRT::Graphics::GraphicsBufferCreateInfo bufferCreateInfo{
175 .cpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Write,
176 .gpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Read,
177 .size = texture2D->GetSize()};
178
179 auto textureStagingBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
180
181 auto texture2DRegion = texture2D->Allocate(texture2D->GetSize(), 4);
182 auto textureStagingBufferRegion = textureStagingBuffer->Allocate(texture2D->GetSize(), 4);
183
184 auto pTextureData = textureStagingBuffer->template Map<uint8_t>(textureStagingBufferRegion);
185
186 NovelRT::Utilities::Memory::Copy(std::span<const uint8_t>(textureMetadata.data), pTextureData);
187
188 textureStagingBuffer->UnmapAndWrite(textureStagingBufferRegion);
189
190 auto gfxContext = _graphicsDevice->CreateGraphicsContext();
191 auto uploadCmdList = gfxContext->CreateCmdList();
192 uploadCmdList->Begin();
193 uploadCmdList->CmdBeginTexturePipelineBarrierLegacyVersion(texture2D);
194 uploadCmdList->CmdCopy(texture2D, textureStagingBufferRegion);
195 uploadCmdList->CmdEndTexturePipelineBarrierLegacyVersion(texture2D);
196 uploadCmdList->End();
197
198 std::vector<std::shared_ptr<NovelRT::Graphics::GraphicsCmdList<TGraphicsBackend>>> lists{
199 uploadCmdList};
200
201 std::vector<std::pair<std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>,
202 uint64_t>>
203 signalSemaphores{std::make_pair(_uploadSemaphore, ++_submittedUploads)};
204
205 _graphicsDevice->QueueSubmit(lists, signalSemaphores);
206
207 Components::TrackedSemaphore<TGraphicsBackend> newUploadTracker{
208 .semaphore = new std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>(
209 _uploadSemaphore),
210 .signalValue = _submittedUploads};
211
212 auto id = completionCatalogue.CreateEntity();
213
214 completionCatalogue.GetComponentView<Components::TrackedSemaphore<TGraphicsBackend>>()
215 .AddComponent(id, newUploadTracker);
216
217 _pendingStagingBuffers.push_back(
218 {_submittedUploads, textureStagingBuffer, gfxContext, uploadCmdList});
219 _textureCache.emplace(
220 assetId, TextureCacheItem{texture2DRegion,
221 Maths::GeoVector2F(static_cast<float>(textureMetadata.width),
222 static_cast<float>(textureMetadata.height))});
223 _pendingLoads.erase(assetId);
224 });
225 }
226 }

228 void SubmitToOrchestrator(Catalogue& catalogue)
229 {
230 auto lastSubmittedFrame = _orchestratorSubmissionSemaphore->GetValue();
231
232 if (lastSubmittedFrame > 0)
233 {
234 lastSubmittedFrame -= 1;
235 }
236
237 _pendingContexts.erase(std::remove_if(_pendingContexts.begin(), _pendingContexts.end(),
238 [lastSubmittedFrame](const auto& pending)
239 { return pending.frameIndex <= lastSubmittedFrame; }),
240 _pendingContexts.end());
241
242 auto [renderPassView, cmdListView, spriteView, transformView, trackedSemaphoresView, cameraView,
243 viewportView] =
244 catalogue.GetComponentViews<
245 Components::RenderPass<TGraphicsBackend>, Components::BuiltCommandList<TGraphicsBackend>,
246 Components::Sprite, NovelRT::Ecs::Components::TransformComponent,
247 Components::TrackedSemaphore<TGraphicsBackend>, Components::Camera, Components::Viewport>();
248
249 std::vector<CameraData> worldSpaceCameras{};
250
251 for (auto [cameraEntityId, camera] : cameraView)
252 {
253 if (camera.isScreenSpace)
254 {
255 continue;
256 }
257
258 Components::Viewport viewport = viewportView.GetComponent(cameraEntityId);
259
260 NovelRT::Ecs::Components::TransformComponent cameraTransform =
261 transformView.GetComponent(cameraEntityId);
262
263 worldSpaceCameras.push_back({cameraEntityId, camera, viewport, cameraTransform});
264
265 if (_cameraConstantBuffers.find(cameraEntityId) == _cameraConstantBuffers.end())
266 {
267 NovelRT::Graphics::GraphicsBufferCreateInfo bufferCreateInfo{
268 .bufferKind = NovelRT::Graphics::GraphicsBufferKind::Constant,
269 .cpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Write,
270 .gpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Read,
271 .size = sizeof(CameraConstantBuffer)};
272
273 auto constantBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
274 auto constantBufferRegion = constantBuffer->Allocate(sizeof(CameraConstantBuffer), 16);
275 _cameraConstantBuffers[cameraEntityId] = constantBufferRegion;
276 }
277 }
278
279 if (worldSpaceCameras.empty())
280 {
281 return; // Not the sprite renderer's job to manage the window or surface
282 }
283
284 auto context = _graphicsDevice->CreateGraphicsContext();
285
286 if (!_orchestratorSubmissionSemaphoreEntity.has_value())
287 {
288 _orchestratorSubmissionSemaphoreEntity = catalogue.CreateEntity();
289 }
290
291 Components::TrackedSemaphore<TGraphicsBackend> signalSemaComponent{};
292 signalSemaComponent.isWaitSemaphore = false;
293 signalSemaComponent.semaphore = new std::shared_ptr<NovelRT::Graphics::GraphicsSemaphore<TGraphicsBackend>>(
294 _orchestratorSubmissionSemaphore);
295 signalSemaComponent.signalValue = ++_frameCounter;
296
297 trackedSemaphoresView.AddComponent(_orchestratorSubmissionSemaphoreEntity.value(), signalSemaComponent);
298
299 _pendingContexts.emplace_back(_frameCounter, context);
300
301 for (auto [entityId, sprite] : spriteView)
302 {
303 auto textureCacheIterator = _textureCache.find(sprite.assetId);
304
305 if (textureCacheIterator == _textureCache.end())
306 {
307 continue;
308 }
309
310 auto& texture = (*textureCacheIterator).second;
311
312 NovelRT::Ecs::Components::TransformComponent transform = transformView.GetComponent(entityId);
313
314 auto currentCmdList =
315 context->CreateCmdList(std::optional<NovelRT::Graphics::SecondaryCmdListInfo<TGraphicsBackend>>(
316 {_renderPass.RenderPass, 0}));
317
318 currentCmdList->Begin();
319 currentCmdList->CmdBindPipeline(_pipeline);
320
321 std::array<
322 std::reference_wrapper<const std::shared_ptr<NovelRT::Graphics::GraphicsBuffer<TGraphicsBackend>>>,
323 1>
324 buffers{std::cref(_vertexBuffer)};
325 std::array<size_t, 1> offsets{_vertexBufferRegion->GetOffset()};
326 currentCmdList->CmdBindVertexBuffers(0, 1, buffers, offsets);
327
328 auto* descriptorSetArray = new std::shared_ptr<
329 NovelRT::Graphics::GraphicsDescriptorSet<TGraphicsBackend>>[worldSpaceCameras.size()];
330 size_t descriptorSetIndex = 0;
331
332 for (auto& cameraData : worldSpaceCameras)
333 {
334
335 auto finalScale = Maths::GeoVector2F(
336 (texture.textureSize.x / static_cast<float>(cameraData.camera.referenceResolutionWidth)) *
337 transform.scale.x,
338 (texture.textureSize.y / static_cast<float>(cameraData.camera.referenceResolutionHeight)) *
339 transform.scale.y);
340
341 auto model = Maths::GeoMatrix4x4F::GetDefaultIdentity();
342 model.Translate(NovelRT::Ecs::Components::TransformComponent::TransformToVector3F(transform, 0.0f));
343 model.Rotate(transform.rotationInRadians);
344 model.Scale(finalScale);
345
346 SpritePushConstant pushConstant{
347 .model = model,
348 .tintColour = Maths::GeoVector4F(sprite.tint.getRScalar(), sprite.tint.getGScalar(),
349 sprite.tint.getBScalar(), sprite.tint.getAScalar())};
350
351 currentCmdList->CmdPushConstants(
352 _pipeline->GetSignature(), NovelRT::Graphics::ShaderProgramVisibility::All, 0,
353 NovelRT::Utilities::Span<uint8_t>(reinterpret_cast<uint8_t*>(&pushConstant),
354 sizeof(SpritePushConstant)));
355
356 NovelRT::Graphics::ViewportInfo viewportInfoStruct{.x = cameraData.viewport.x,
357 .y = cameraData.viewport.y +
358 cameraData.viewport.height,
359 .width = cameraData.viewport.width,
360 .height = -(cameraData.viewport.height),
361 .minDepth = 0.0f,
362 .maxDepth = 1.0f};
363
364 currentCmdList->CmdSetViewport(viewportInfoStruct);
365 currentCmdList->CmdSetScissor(
366 Components::Viewport::ViewportLocationToVector2F(cameraData.viewport),
367 Components::Viewport::ViewportDimensionsToVector2F(cameraData.viewport));
368
369 auto view = cameraData.camera.CreateViewMatrix(cameraData.transform);
370
371 auto projection = cameraData.camera.CreateProjectionMatrix(cameraData.camera);
372
373 auto& constantBufferRegion = _cameraConstantBuffers.at(cameraData.entityId);
374 auto pCameraData = constantBufferRegion->template Map<CameraConstantBuffer>();
375 pCameraData[0].view = view;
376 pCameraData[0].projection = projection;
377 constantBufferRegion->UnmapBytesAndWrite();
378
379 std::vector<std::shared_ptr<NovelRT::Graphics::GraphicsResourceMemoryRegion<
380 NovelRT::Graphics::GraphicsResource, TGraphicsBackend>>>
381 inputRegions{texture.textureRegion, constantBufferRegion};
382
383 auto descriptorSet = _pipeline->CreateDescriptorSet();
384 descriptorSet->AddMemoryRegionsToInputs(inputRegions);
385 descriptorSet->UpdateDescriptorSetData();
386
387 descriptorSetArray[descriptorSetIndex] = descriptorSet;
388
389 currentCmdList->CmdBindDescriptorSet(descriptorSet);
390
391 descriptorSetIndex++;
392
393 currentCmdList->CmdDraw(6, 1, 0, 0);
394 }
395
396 currentCmdList->End();
397
398 Components::RenderPass<TGraphicsBackend> passComponent{.renderPassIndex = _renderPass.RenderPassId,
399 .descriptorSets = descriptorSetArray,
400 .descriptorSetCount = worldSpaceCameras.size()};
401
402 Components::BuiltCommandList<TGraphicsBackend> cmdListComp{
403 .commandList =
404 new std::shared_ptr<NovelRT::Graphics::GraphicsCmdList<TGraphicsBackend>>(currentCmdList)};
405
406 renderPassView.AddComponent(entityId, passComponent);
407 cmdListView.AddComponent(entityId, cmdListComp);
408 }
409 }

412 SpriteRendererSystem(
413 std::shared_ptr<NovelRT::Graphics::GraphicsDevice<TGraphicsBackend>> device,
414 SpritePass renderPass,
415 std::shared_ptr<ResourceManagement::Desktop::DesktopResourceLoader> resourceLoader,
416 std::shared_ptr<NovelRT::Graphics::GraphicsMemoryAllocator<TGraphicsBackend>> memoryAllocator,
417 std::shared_ptr<NovelRT::Graphics::GraphicsSurfaceContext<TGraphicsBackend>> surfaceContext)
418 : _resourceLoader(std::move(resourceLoader)),
419 _graphicsDevice(std::move(device)),
420 _memoryAllocator(std::move(memoryAllocator)),
421 _renderPass(std::move(renderPass)),
422 _surfaceContext(std::move(surfaceContext)),
423 _uploadSemaphore(_graphicsDevice->CreateSemaphore(0)),
424 _orchestratorSubmissionSemaphore(_graphicsDevice->CreateSemaphore(0))
425 {
426 NovelRT::Graphics::GraphicsBufferCreateInfo bufferCreateInfo{
427 .cpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Write,
428 .gpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Read,
429 .size = 64 * 1024};
430
431 auto stagingBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
432
433 bufferCreateInfo.bufferKind = NovelRT::Graphics::GraphicsBufferKind::Vertex;
434 bufferCreateInfo.cpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::None;
435 bufferCreateInfo.gpuAccessKind = NovelRT::Graphics::GraphicsResourceAccess::Write;
436
437 _vertexBuffer = _memoryAllocator->CreateBuffer(bufferCreateInfo);
438 _vertexBufferRegion = _vertexBuffer->Allocate(sizeof(SpriteVertexShaderInputs) * 6, 16);
439 auto stagingRegion = stagingBuffer->Allocate(sizeof(SpriteVertexShaderInputs) * 6, 16);
440
441 auto pVertexBuffer = stagingBuffer->template Map<SpriteVertexShaderInputs>(stagingRegion);
442
443 pVertexBuffer[0] =
444 SpriteVertexShaderInputs{Maths::GeoVector3F(-1.f, 1.f, 0.0f), Maths::GeoVector2F(0.0f, 0.0f)};
445 pVertexBuffer[1] =
446 SpriteVertexShaderInputs{Maths::GeoVector3F(1.f, 1.f, 0.0f), Maths::GeoVector2F(1.0f, 0.0f)};
447 pVertexBuffer[2] =
448 SpriteVertexShaderInputs{Maths::GeoVector3F(-1.f, -1.f, 0.0f), Maths::GeoVector2F(0.0f, 1.0f)};
449 pVertexBuffer[3] =
450 SpriteVertexShaderInputs{Maths::GeoVector3F(-1.f, -1.f, 0.0f), Maths::GeoVector2F(0.0f, 1.0f)};
451 pVertexBuffer[4] =
452 SpriteVertexShaderInputs{Maths::GeoVector3F(1.f, 1.f, 0.0f), Maths::GeoVector2F(1.0f, 0.0f)};
453 pVertexBuffer[5] =
454 SpriteVertexShaderInputs{Maths::GeoVector3F(1.f, -1.f, 0.0f), Maths::GeoVector2F(1.0f, 1.0f)};
455
456 stagingBuffer->UnmapAndWrite(stagingRegion);
457
458 auto vertShaderData = _resourceLoader->LoadShaderSource("SpriteVertex.spv");
459 auto pixelShaderData = _resourceLoader->LoadShaderSource("SpritePixel.spv");
460
461 std::vector<NovelRT::Graphics::GraphicsPipelineInputElement> elements{
462 NovelRT::Graphics::GraphicsPipelineInputElement(
463 typeid(Maths::GeoVector3F), NovelRT::Graphics::GraphicsPipelineInputElementKind::Position, 12),
464 NovelRT::Graphics::GraphicsPipelineInputElement(
465 typeid(Maths::GeoVector2F), NovelRT::Graphics::GraphicsPipelineInputElementKind::TextureCoordinate,
466 8)};
467
468 std::vector<NovelRT::Graphics::GraphicsPipelineInput> inputs{
469 NovelRT::Graphics::GraphicsPipelineInput(elements)};
470
471 std::vector<NovelRT::Graphics::GraphicsPipelineResource> resources{
472 NovelRT::Graphics::GraphicsPipelineResource(NovelRT::Graphics::GraphicsPipelineResourceKind::Texture,
473 NovelRT::Graphics::ShaderProgramVisibility::Pixel),
474 NovelRT::Graphics::GraphicsPipelineResource(
475 NovelRT::Graphics::GraphicsPipelineResourceKind::ConstantBuffer,
476 NovelRT::Graphics::ShaderProgramVisibility::Vertex)};
477
478 std::vector<NovelRT::Graphics::GraphicsPushConstantRange> pushConstantRanges{
479 NovelRT::Graphics::GraphicsPushConstantRange{NovelRT::Graphics::ShaderProgramVisibility::All, 0,
480 sizeof(SpritePushConstant)}};
481
482 auto signature = _graphicsDevice->CreatePipelineSignature(
483 NovelRT::Graphics::GraphicsPipelineBlendFactor::SrcAlpha,
484 NovelRT::Graphics::GraphicsPipelineBlendFactor::OneMinusSrcAlpha, inputs, resources,
485 NovelRT::Utilities::Span<NovelRT::Graphics::GraphicsPushConstantRange>(pushConstantRanges));
486
487 auto vertShaderProg = _graphicsDevice->CreateShaderProgram(
488 "main", NovelRT::Graphics::ShaderProgramKind::Vertex, vertShaderData.shaderCode);
489 auto pixelShaderProg = _graphicsDevice->CreateShaderProgram(
490 "main", NovelRT::Graphics::ShaderProgramKind::Pixel, pixelShaderData.shaderCode);
491
492 _pipeline =
493 _graphicsDevice->CreatePipeline(signature, vertShaderProg, pixelShaderProg, _renderPass.RenderPass);
494
495 auto gfxContext = _graphicsDevice->CreateGraphicsContext();
496 auto cmdList = gfxContext->CreateCmdList();
497 cmdList->Begin();
498 cmdList->CmdCopy(_vertexBufferRegion, stagingRegion);
499 cmdList->End();
500
501 std::vector<std::shared_ptr<NovelRT::Graphics::GraphicsCmdList<TGraphicsBackend>>> lists{cmdList};
502
503 _graphicsDevice->QueueSubmit(lists);
504 _graphicsDevice->WaitForIdle();
505 }
